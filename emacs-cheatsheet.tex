\medskip
\textbf{Example:} Adding a keybind to save and load cursor position
from register.

\begin{verbatim}
;;; ~/.emacs.d/init.el
(global-set-key
 (kbd "C-c s")
 (lambda ()
   (interactive)
   (point-to-register 'a)))
(global-set-key
 (kbd "C-c f")
 (lambda ()
   (interactive)
   (jump-to-register 'a)))
\end{verbatim}
